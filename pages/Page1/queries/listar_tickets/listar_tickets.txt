SELECT 
  t.id,
  t.nome_cliente,
  t.cpf,
  t.uf,
  t.numero_venda,
  CASE COALESCE(t.canal_marketplace, 'nao_informado')
    WHEN 'mercado_livre' THEN 'Mercado Livre'
    WHEN 'shopee' THEN 'Shopee'
    WHEN 'magalu' THEN 'Magalu'
    WHEN 'amazon' THEN 'Amazon'
    WHEN 'site' THEN 'Site'
    ELSE 'Não informado'
  END AS canal_marketplace,
  CASE t.empresa::TEXT
    WHEN 'lessul' THEN 'Lessul'
    WHEN 'ms_decor' THEN 'MS Decor'
    WHEN 'viva_vida' THEN 'Viva Vida'
    WHEN 'movelbento' THEN 'Movelbento'
    WHEN 'modifika' THEN 'Modifika'
    ELSE COALESCE(t.empresa::TEXT, '-')
  END AS empresa,
  t.produto,
  t.sku,
  CASE t.motivo::TEXT
    WHEN 'desistencia' THEN 'Desistência'
    WHEN 'defeito_fabricacao' THEN 'Defeito de Fabricação'
    WHEN 'produto_incorreto' THEN 'Produto incorreto'
    WHEN 'faltando_itens' THEN 'Faltando itens'
    WHEN 'produto_danificado' THEN 'Produto danificado'
    WHEN 'problema' THEN 'Problema'
    ELSE COALESCE(t.motivo::TEXT, '-')
  END AS motivo,
  CASE t.status_reclamacao::TEXT
    WHEN 'afetando' THEN 'Afetando'
    WHEN 'nao_afetando' THEN 'Não afetando'
    WHEN 'removida' THEN 'Removida'
    ELSE COALESCE(t.status_reclamacao::TEXT, '-')
  END AS status_reclamacao,
  CASE t.status_ticket::TEXT
    WHEN 'aberto' THEN 'Aberto'
    WHEN 'concluido' THEN 'Concluído'
    WHEN 'aguardando_cliente' THEN 'Aguardando cliente'
    WHEN 'aguardando_devolucao' THEN 'Aguardando devolução'
    WHEN 'aguardando_assistencia' THEN 'Aguardando assistência'
    WHEN 'aguardando_marketplace' THEN 'Aguardando marketplace'
    ELSE COALESCE(t.status_ticket::TEXT, '-')
  END AS status_ticket,
  CASE t.sla_status
    WHEN 'concluido' THEN 'Concluído'
    WHEN 'atrasado' THEN 'Atrasado'
    WHEN 'no_prazo' THEN 'No prazo'
    ELSE COALESCE(t.sla_status, '-')
  END AS sla_status,
  t.prazo_conclusao,
  t.custos_totais,
  t.valor_reembolso,
  t.valor_coleta,
  t.data_reclamacao,
  t.criado_em,
  p.nome AS responsavel_nome
FROM public.vw_tickets t
LEFT JOIN public.perfis p
  ON p.id = t.responsavel
WHERE t.ativo = TRUE

  AND (
    {{ !inp_busca_nome.text ? "TRUE" : "FALSE" }}
    OR LOWER(t.nome_cliente) LIKE LOWER({{ "'%" + inp_busca_nome.text.trim().replace(/'/g, "''") + "%'" }})
  )

  AND (
    {{ !sel_empresa.selectedOptionValue ? "TRUE" : "FALSE" }}
    OR t.empresa::TEXT = {{ "'" + sel_empresa.selectedOptionValue + "'" }}
  )

  AND (
    {{ !sel_status_ticket.selectedOptionValue ? "TRUE" : "FALSE" }}
    OR t.status_ticket::TEXT = {{ "'" + sel_status_ticket.selectedOptionValue + "'" }}
  )

  AND (
    {{ !sel_motivo.selectedOptionValue ? "TRUE" : "FALSE" }}
    OR t.motivo::TEXT = {{ "'" + sel_motivo.selectedOptionValue + "'" }}
  )

  AND (
    {{ !sel_status_reclamacao.selectedOptionValue ? "TRUE" : "FALSE" }}
    OR t.status_reclamacao::TEXT = {{ "'" + sel_status_reclamacao.selectedOptionValue + "'" }}
  )

  AND (
    {{ !sel_canal.selectedOptionValue ? "TRUE" : "FALSE" }}
    OR t.canal_marketplace = {{ "'" + sel_canal.selectedOptionValue + "'" }}
  )

  AND (
    {{ !sel_responsavel.selectedOptionValue ? "TRUE" : "FALSE" }}
    OR t.responsavel = {{ "'" + sel_responsavel.selectedOptionValue + "'::UUID" }}
    OR t.criado_por  = {{ "'" + sel_responsavel.selectedOptionValue + "'::UUID" }}
  )

  AND (
    {{ !dp_data_inicio.selectedDate ? "TRUE" : "FALSE" }}
    OR COALESCE(t.data_reclamacao, t.criado_em::date) >= {{ "'" + moment(dp_data_inicio.selectedDate).format("YYYY-MM-DD") + "'" }}
  )

  AND (
    {{ !dp_data_fim.selectedDate ? "TRUE" : "FALSE" }}
    OR COALESCE(t.data_reclamacao, t.criado_em::date) <= {{ "'" + moment(dp_data_fim.selectedDate).format("YYYY-MM-DD") + "'" }}
  )

ORDER BY t.criado_em DESC;
