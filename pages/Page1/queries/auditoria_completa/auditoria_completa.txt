SELECT
  a.id,
  a.ticket_id,
  a.acao,
  a.campo,
  a.valor_antigo,
  a.valor_novo,
  a.usuario_nome,
  a.data_hora
FROM public.ticket_auditoria a
WHERE 
  -- Filtro de Início: O JS avalia. Se tiver data, usa ela. Se não, usa '1970-01-01'
  a.data_hora >= ({{ dp_aud_inicio.selectedDate ? dp_aud_inicio.selectedDate : '1970-01-01' }})::TIMESTAMPTZ
  
  -- Filtro de Fim: Mesmo princípio. Somamos 1 dia para pegar até 23:59:59 do dia selecionado
  AND a.data_hora < ({{ dp_aud_fim.selectedDate ? dp_aud_fim.selectedDate : '2099-12-31' }})::DATE + INTERVAL '1 day'

ORDER BY a.data_hora DESC

-- Nota de arquitetura: LIMIT fixo é OK para essa tela no MVP, mas na lista de tickets use Paginação.
LIMIT 500;
