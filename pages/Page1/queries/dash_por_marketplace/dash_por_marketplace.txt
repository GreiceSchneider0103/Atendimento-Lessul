-- [Query: dash_por_marketplace]
-- Gráfico: Reclamações por Canal (valores padronizados)
SELECT
  COALESCE(public.fn_normalizar_canal_marketplace(canal_marketplace), 'nao_informado') AS canal_marketplace,
  COUNT(*)                        AS qtd,
  COALESCE(SUM(custos_totais), 0) AS custo
FROM public.tickets
WHERE ativo = TRUE
GROUP BY 1
ORDER BY qtd DESC;
