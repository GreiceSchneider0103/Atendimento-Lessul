SELECT
  COUNT(*) FILTER (WHERE t.ativo = TRUE) AS total_tickets,
  COUNT(*) FILTER (WHERE t.ativo = TRUE AND t.status_ticket <> 'concluido') AS tickets_abertos,
  COUNT(*) FILTER (WHERE t.ativo = TRUE AND t.sla_status = 'atrasado') AS tickets_atrasados,
  COALESCE(SUM(t.custos_totais) FILTER (WHERE t.ativo = TRUE), 0) AS custo_total,
  COALESCE(SUM(t.valor_reembolso) FILTER (WHERE t.ativo = TRUE), 0) AS valor_total_reembolso,
  COALESCE(SUM(t.valor_coleta) FILTER (WHERE t.ativo = TRUE), 0) AS total_coleta
FROM public.vw_tickets t;
