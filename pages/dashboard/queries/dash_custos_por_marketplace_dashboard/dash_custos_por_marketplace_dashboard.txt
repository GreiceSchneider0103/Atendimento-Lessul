SELECT
  CASE COALESCE(canal_marketplace, 'nao_informado')
    WHEN 'mercado_livre' THEN 'Mercado Livre'
    WHEN 'shopee' THEN 'Shopee'
    WHEN 'magalu' THEN 'Magalu'
    WHEN 'amazon' THEN 'Amazon'
    WHEN 'site' THEN 'Site'
    ELSE 'Não informado'
  END AS marketplace,
  COALESCE(SUM(custos_totais), 0) AS custo
FROM public.tickets
WHERE ativo = TRUE
GROUP BY COALESCE(canal_marketplace, 'nao_informado')
ORDER BY custo DESC;
